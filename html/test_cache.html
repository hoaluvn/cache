 56 def test_entry_invalidate():
 57     my_mem = memory("database.yml")
 58     cache1 = cache(mem=my_mem)
 59     cache2 = cache(mem=my_mem)
 60     raddr = int(random.random()* (2 ** 14)) * 4
 61     rdata = int(random.random() * (2 ** 32))
 62     # first write
 63     cache1.cache_write(raddr, rdata)
 64     assert rdata == cache1.cache_read(raddr)
 65     # clean entry
 66     cache1.write_reg(1, raddr)
 67     cache1.write_reg(2, 4)
 68     cache1.write_reg(0, 2)
 69     # second write
 70     rdata2 = int(random.random() * (2 ** 32))
 71     cache1.cache_write(raddr, rdata2)
 72     assert rdata2 == cache1.cache_read(raddr)
 73     # invalidate
 74     cache1.write_reg(1, raddr)
 75     cache1.write_reg(2, 4)
 76     cache1.write_reg(0, 1)
 77     # read the first write
 78     assert rdata == cache1.cache_read(raddr)

 80 def test_fillup():
 81     my_mem = memory("database.yml")
 82     cache1 = cache(mem=my_mem)
 83     cache2 = cache(mem=my_mem)
 84     table = {}
 85     baddr = int(random.random()* (2 ** 14)) * 4
 86     # Fill up cache array
 87     for i in range(8):
 88         raddr = baddr + i*4 + i * 32 * random.randint(1,32)
 89         rdata = int(random.random() * (2 ** 32))
 90         table[raddr] = rdata
 91         cache1.cache_write(raddr, rdata)
 92     # Clean entry from baddr => +32*8
 93     cache1.write_reg(1, baddr)
 94     cache1.write_reg(2, 32*8)
 95     cache1.write_reg(0, 2)
 96     # Read back and check
 97     for addr in table:
 98         data = table[addr]
 99         if baddr <= addr <= baddr + 32*8:
100             assert data == cache2.cache_read(addr)
101         else:
102             assert data != cache2.cache_read(addr)
