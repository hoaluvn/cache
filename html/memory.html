1 import yaml
 2 import logging
 3 from random import random
 4 
 5 class memory:
 6     """
 7     Main memory model
 8        storage_path  : "secondary" storage
 9        addr_width    :
10        byte_align    :
11        data_width    :
12     """
13     def __init__(self, storage_path="db.yml", addr_width=16, byte_align=4, data_width=32):
14         self.storage    = storage_path
15         self.addr_width = int(addr_width)
16         self.byte_align = int(byte_align)
17         self.data_width = int(data_width)
18         self.mem_size   = 2 ** self.addr_width
19         self.db = {}
20 
21         """ Load content from secondary storage """
22         try:
23             with open(self.storage, "r") as ymlfile:
24                 self.db = yaml.load(ymlfile, Loader=yaml.BaseLoader)
25         except IOError:
26             logging.warning("DB file not exist, will create a new one")
27 
28     def __del__(self):
29         self.mem_flush()
30 
31     def __addr_check(self, addr):
32         res = True
33         if addr & (self.byte_align - 1):
34             logging.error("Address alignment: 0x%x", addr)
35             res = False
36         if addr > self.mem_size - self.byte_align:
37             logging.error("Address is out of range: 0x%x", addr)
38             res = False
39         return res
40 
41     def mem_read(self, addr):
42         if self.__addr_check(addr):
43             hexaddr = hex(addr)
44             if hexaddr not in self.db:
45                 logging.warning("Read uninitialized address: 0x%x", addr)
46                 data = int(random() * (2 ** self.data_width - 1) )
47                 self.db[hexaddr] = hex(data)
48             else:
49                 data = int(self.db[hexaddr], 0)
50             logging.info("Memmory read : [0x%04x] => 0x%x", addr, data)
51             return data
52 
53     def mem_write(self, addr, data):
54         if self.__addr_check(addr):
55             logging.info("Memmory write: [0x%04x] <= 0x%x", addr, data)
56             self.db[hex(addr)] = hex(data)
57 
58     """ Store content to secondary storage """
59     def mem_flush(self):
60         with open(self.storage, "w") as ymlfile:
61             yaml.dump(self.db, ymlfile)
