1 import logging

  3 class cacherow:
  4     def __init__(self, index, addrw, tagw, indexw, dataw, mem=None):
  5         self.idx = index
  6         self.aw = addrw
  7         self.tw = tagw
  8         self.iw = indexw
  9         self.dw = dataw
 10         self.dirty = False
 11         self.valid = False
 12         self.tag   = 0
 13         self.data  = 0
 14         self.mem = mem
 15         self.addrmask = (2 ** addrw - 1) & ~(2 ** (addrw - tagw - indexw) - 1)
 16 
 17     def __get_tag(self, addr):
 18         _tag = addr >> (self.aw - self.tw)
 19         logging.debug("addr: 0x%x => tag: 0x%x", addr, _tag)
 20         return _tag
 21 
 22     def __get_addr(self, tag):
 23         a = self.tag << (self.aw - self.tw) 
 24         b = self.idx << (self.aw - self.tw - self.iw)
 25         _addr = a + b
 26         logging.debug("tag: 0x%x => addr: 0x%x", tag, _addr)
 27         return _addr
 28 
 29     def get_addr(self):
 30         if self.valid:
 31             return self.__get_addr(self.tag)
 32 
 33     def invalid_entry(self):
 34         self.valid = False
 35         self.dirty = False
 36         logging.info("Cache invalidate entry: %d", self.idx)
 37 
 38     def clean_entry(self):
 39         if self.valid and self.dirty:
 40             _addr = self.__get_addr(self.tag)
 41             self.mem.mem_write(_addr, self.data)
 42             self.dirty = False
 43             logging.info("Flush cache line to memory 0x%x => [0x%x]", self.data, _addr)
 44 
 45     def store(self, addr, data):
 46         tag   = self.__get_tag(addr)
 47         if self.valid and self.dirty and self.tag != tag:
 48             self.clean_entry()
 49         self.tag   = tag
 50         self.dirty = True
 51         self.valid = True
 52         self.data  = data & (2 ** self.dw - 1)
 53 
 54     def load(self, addr):
 55         _tag = self.__get_tag(addr)
 56         if self.valid:
 57             if self.tag == _tag:
 58                 return self.data
 59             elif self.dirty:
 60                 _addr = self.__get_addr(self.tag)
 61                 self.mem.mem_write(_addr, self.data)
 62         self.tag = _tag
 63         self.valid = True
 64         self.dirty = False
 65         self.data  = self.mem.mem_read(addr & self.addrmask)
 66         return self.data

 68 class cache:
 69     def __init__(self, addrw=16, tagw=11, idxw=3, dataw=32, mem=None):
 70         self.aw = addrw
 71         self.tw = tagw
 72         self.iw = idxw
 73         self.dw = dataw
 74         self.sz = 2 ** idxw
 75         self.reg = [0,0,0]
 76         self.cachearray = {}
 77         for i in range(self.sz):
 78             self.cachearray[i] = cacherow(i, self.aw, self.tw, self.iw, self.dw, mem)
 79 
 80     def __get_index(self, addr):
 81         _index = (addr >> (self.aw - self.tw - self.iw)) & (2 ** self.iw - 1)
 82         logging.debug("index: %d", _index)
 83         return _index
 84 
 85     def cache_write(self, addr, data):
 86         logging.info("Cache write start: [0x%04x] <= 0x%x", addr, data)
 87         idx = self.__get_index(addr)
 88         self.cachearray[idx].store(addr, data)
 89         logging.info("Cache write end")
 90 
 91     def cache_read(self, addr):
 92         logging.info("Cache read start: [0x%04x]", addr)
 93         idx = self.__get_index(addr)
 94         _data = self.cachearray[idx].load(addr)
 95         logging.info("Cache read end  : [0x%04x] => 0x%x", addr, _data)
 96         return _data
 97 
 98     def write_reg(self, rid, data):
 99         self.reg[rid] = data & (2 ** self.aw - 1)
100         logging.debug("Register write [%d] = 0x%x", rid, self.reg[rid])
101         if rid == 0: # Execute command
102             cmd = data & 3
103             if cmd == 0:
104                 return
105             start_addr = self.reg[1]
106             end_addr   = start_addr + self.reg[2]
107             logging.debug("Test start_addr: 0x%x", start_addr)
108             logging.debug("Test end_addr: 0x%x", end_addr)
109             for i in range(self.sz):
110                 entry_addr = self.cachearray[i].get_addr()
111                 if self.cachearray[i].valid:
112                     logging.debug("Checking entry address: entry[%d]:addr=0x%x", i, entry_addr)
113                     if start_addr <= entry_addr <= end_addr:
114                         if cmd == 3: # clean & invalidate
115                             self.cachearray[i].clean_entry()
116                             self.cachearray[i].invalid_entry()
117                         elif cmd == 2: # clean entires
118                             self.cachearray[i].clean_entry()
119                         elif cmd == 1: # invalidate entries
120                             self.cachearray[i].invalid_entry()
