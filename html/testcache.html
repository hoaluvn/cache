 21 class Values:
 22     ''' Storage class '''
 23     def __init__(self, defaults=None):
 24         if defaults:
 25             for (attr, val) in list(defaults.items()):
 26                 setattr(self, attr, val)

 28 def get_logging_level(level):
 29     """ string to logging.Level """
 30     selector = logging.WARNING
 31     switcher = {"info"   : logging.INFO,
 32                 "warning": logging.WARNING,
 33                 "debug"  : logging.DEBUG
 34                }
 35     if level in switcher:
 36         selector = switcher[level]
 37     return selector
 38 
 39 def parse_options():
 40     """ Parse command line args """
 41     m_args = sys.argv[1:]
 42     m_options = Values({
 43         "seed": None,
 44         "has_logfile" : None,
 45         "logfile" : "log",
 46         "loglevel": "info",
 47     })
 48     while m_args:
 49         if m_args[0] == '--seed':
 50             m_options.seed = m_args[1]
 51             m_args = m_args[2:]
 52         elif m_args[0] == '-l':
 53             m_options.has_logfile = 1
 54             m_options.logfile = m_args[1]
 55             m_args = m_args[2:]
 56         elif m_args[0] == '-d':
 57             m_options.loglevel = m_args[1]
 58             m_args = m_args[2:]
 59         elif m_args[0] in ('-h', '--help'):
 60             usage()
 61             sys.exit(0)
 62         elif m_args[0][0] == "-":
 63             print("Unknown option", m_args[0], file=sys.stderr)
 64             print(usage, file=sys.stderr)
 65             exit(0)
 66         else:
 67             break
 68     return (m_options, m_args)
